    Type declaration

    • all: <T, E>(results: Result<T, E>[]) => Result<T[], E>

      Converts an array of Results into a Result of array. Fails fast on the first error encountered.

      Example

      const results = [ok(1), ok(2), ok(3)];
      const combined = batch.all(results);
      // Returns: Ok([1, 2, 3])

      const mixed = [ok(1), err("failed"), ok(3)];
      const failed = batch.all(mixed);
      // Returns: Err("failed")
    • allAsync: <T, E>(promises: Promise<Result<T, E>>[]) => Promise<Result<T[], E>>

      Waits for all Promise to settle, then combines like batch.all().

      Example

      const promises = [fetchUser(1), fetchUser(2), fetchUser(3)];
      const result = await batch.allAsync(promises);
      // Returns: Ok([user1, user2, user3]) or first Err
    • allSettledAsync: <T, E>(promises: Promise<Result<T, E>>[]) => Promise<{ oks: T[]; errors: E[] }>

      Waits for all Promise to settle and partitions successes from errors. Unlike allAsync, this never fails and returns both successes and errors.

      Example

      const promises = [fetchUser(1), fetchUser(2), fetchUser(3)];
      const { oks, errors } = await batch.allSettledAsync(promises);
      console.log(`Loaded ${oks.length} users, ${errors.length} failed`);
    • oks: <T, E>(results: Result<T, E>[]) => T[]

      Extracts all success values from an array of Results. Performance: ~2-3x faster than filter().map() chains.

      Example

      const results = [ok(1), err("failed"), ok(3), err("error")];
      const successes = batch.oks(results);
      // Returns: [1, 3]
    • errs: <T, E>(results: Result<T, E>[]) => E[]

      Extracts all error values from an array of Results. Performance: ~2-3x faster than filter().map() chains.

      Example

      const results = [ok(1), err("failed"), ok(3), err("error")];
      const errors = batch.errs(results);
      // Returns: ["failed", "error"]
    • partition: <T, E>(results: Result<T, E>[]) => { oks: T[]; errors: E[] }

      Separates an array of Results into successes and errors.

      Example

      const results = [ok(1), err("failed"), ok(3)];
      const { oks, errors } = batch.partition(results);
      // oks: [1, 3], errors: ["failed"]
    • partitionWith: <T, E>(
          results: Result<T, E>[],
      ) => {
          oks: T[];
          errors: E[];
          okCount: number;
          errorCount: number;
          total: number;
      }

      Enhanced partition that includes metadata in a single pass. Performance: Much faster than calling partition() + length calculations separately.

      Example

      const results = [ok(1), err("failed"), ok(3)];
      const stats = batch.partitionWith(results);
      // stats.oks: [1, 3], stats.errors: ["failed"]
      // stats.okCount: 2, stats.errorCount: 1, stats.total: 3
    • analyze: <T, E>(
          results: Result<T, E>[],
      ) => {
          okCount: number;
          errorCount: number;
          total: number;
          hasErrors: boolean;
          isEmpty: boolean;
      }

      Analyzes an array of Results without extracting values for maximum performance. Single-pass analysis that's much faster than multiple operations.

      Example

      const results = [ok(1), err("failed"), ok(3)];
      const stats = batch.analyze(results);
      // stats.okCount: 2, stats.errorCount: 1, stats.total: 3
      // stats.hasErrors: true, stats.isEmpty: false

      console.log(`Success rate: ${stats.okCount}/${stats.total}`);
      if (stats.hasErrors) console.log("Some operations failed");

      ) => {
          firstOk: undefined | T;

          okIndex: number;
          errorIndex: number;
      }

      Finds the first success and first error with early-exit optimization. Much more efficient than filtering when you only need the first of each type.

      Example

      const results = [err("e1"), ok("success"), err("e2"), ok("ok2")];
      const { firstOk, firstError, okIndex, errorIndex } = batch.findFirst(results);
      // firstOk: "success", firstError: "e1", okIndex: 1, errorIndex: 0

      Reduces an array of Results to a single value with custom handling for successes and errors.

      Example

      const results = [ok(5), err("failed"), ok(10)];
      const sum = batch.reduce(results, {
        onOk: (acc, value) => acc + value,
        onErr: (acc, error) => acc // ignore errors
      }, 0);
      // Returns: 15
    • first: <T, E>(results: Result<T, E>[]) => Result<T, E[]>

      Returns the first successful Result or a Result containing all errors.

      Example

      const results = [err("e1"), err("e2"), ok("success")];
      const first = batch.first(results);
      // Returns: Ok("success")

      const allFailed = [err("e1"), err("e2")];
      const none = batch.first(allFailed);
      // Returns: Err(["e1", "e2"])
